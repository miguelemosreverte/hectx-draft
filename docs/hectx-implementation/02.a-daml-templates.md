**Abstract**
Key Daml templates used for minting and holdings.

**Body**
I. HectX Holding (Splice Holding interface)
```
module HectX.Holding where

template HectXHolding
  with
    owner : Party
    issuer : Party
    instrumentId : Api.Token.HoldingV1.InstrumentId
    amount : Decimal
    mintedAt : Time
    navPerTokenAtMint : Decimal
  where
    signatory owner, issuer

    interface instance Api.Token.HoldingV1.Holding for HectXHolding where
      view = Api.Token.HoldingV1.HoldingView with
        owner
        instrumentId = instrumentId
        amount = amount
        lock = None
        meta = Api.Token.MetadataV1.Metadata
          with values = TextMap.fromList
            [ ("mintedAt", show mintedAt)
            , ("navPerToken", show navPerTokenAtMint)
            ]
```
