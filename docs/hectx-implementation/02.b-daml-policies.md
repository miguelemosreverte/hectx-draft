**Abstract**
Policy templates that govern eligibility and fees.

**Body**
I. Eligibility Policy
```
module HectX.Policy where

template EligibilityPolicy
  with
    admin : Party
    prohibitedJurisdictions : [Text]
    restrictedJurisdictions : TextMap.TextMap RestrictedRule
  where
    signatory admin
```

II. Fee And Mint Policy
```
module HectX.Policy where

template FeeSchedule
  with
    admin : Party
    subscriptionFeeBps : Decimal
    conversionFeeBps : Decimal
    managementFeeBps : Decimal
  where
    signatory admin


template MintPolicy
  with
    admin : Party
    mintingEnabled : Bool
    elasticFeeActive : Bool
    elasticFeeBps : Decimal
    maxNavAgeSeconds : Int
  where
    signatory admin
```
